15def build(eventlog: PysparkLog, 

16 cube_dimensions: Union[Text, Sequence[Text]], 

17 make_start_end: bool = False, 

18 variant: Text = 'frequency', 

19 agg_func: Callable[[Union[Text, Column]], Column] = F.avg, 

20 date_format: Text = 'yyyy-MM-dd HH:mm:ss', 

21 pm4py_params: Optional[Mapping] = None): 

22 """ 

23 Creates either a frequency DFG or a performance DFG for every unique value of the columns provided 

24 in cube_dimensions. The DFG that represents a given value of of cube_dimension is the DFG that is 

25 calculated on all the rows where cube_dimension has that value. The result is returned in the form of 

26 a dict with as keys the cube_dimension values and as values the corresponding DFGs. 

27 

28 Parameters 

29 ---------- 

30 eventlog 

31 The log in the form of a pm4pyspark.log.PysparkLog. 

32 cube_dimensions 

33 A column of the log that contains values of categorical type. One DFG is generated per unique 

34 value of these columns. 

35 make_start_end 

36 A boolean parameter that indicates whether or not to additionally create and return: 

37 - a dict with counts of the start activities 

38 - a dict with counts of the end activities 

39 These respectively are equivalent to what can be obtained with 

40 log.get_start_activities and log.get_end_activities. 

41 Calling this method with `make_start_end=True` is computationally more efficient than 

42 calling it with `make_start_end=False` and running log.get_start_activities and 

43 log.get_end_activities separately. 

44 variant 

45 A string value of either 'frequency' or 'performance', indicating whether to run 

46 pm4pyspark.discovery.dfg.frequency or pm4pyspark.discovery.dfg.performance. 

47 agg_func 

48 Only for variant='performance'. The aggregation function used to aggregate the time differences 

49 between the activity pairs. 

50 date_format 

51 Only for variant='performance'. The date format used to parse the timestamp column. 

52 pm4py_params 

53 Possible parameters passed to the algorithms: 

54 window -> the window size for the DFG. Defaults to 1. 

55 

56 Returns 

57 ------- 

58 result 

59 A pm4pyspark.dfg.frequency.DfgResult object consisting of: 

60 - result.dfg 

61 The DFG graph 

62 - start_activities 

63 A dict with the counts of how frequently each activity occurs at start of a trace, 

64 or `None` when `make_start_end=False` 

65 - end_activities 

66 A dict with the counts of how frequently each activity occurs at end of a trace, or 

67 or `None` when `make_start_end=False` 

68 """

97def _process_dimension_value_combination(row: List[Row], 

98 eventlog: PysparkLog, 

99 params: CubeParams): 

100 assert isinstance(params, CubeParams), 'argument params must be of type CubeParams but is {}'.format(type(params)) 

101 filtered_data = _filter_on_dimensions(eventlog.data_frame, params.cube_dimensions, row) 

102 filtered_pyspark_log = eventlog.copy_with_new_data(filtered_data) 

103 

104 if params.variant == 'frequency': 

105 result = frequency.build(filtered_pyspark_log, 

106 make_start_end=params.make_start_end, 

107 pm4py_params=params.pm4py_params) 

108 elif params.variant == 'performance': 

109 result = performance.build(filtered_pyspark_log, 

110 agg_func=params.agg_func, 

111 date_format=params.date_format, 

112 pm4py_params=params.pm4py_params) 

113 else: 

114 raise ValueError('variant should be either `frequency` or `performance`!') 

115 return result

119 cube_dimensions: Union[Text, Sequence[Text]],

126def _filter_on_dimension(filtered_data: DataFrame, 

127 dimension: Text, 

128 value: Optional[Any]): 

129 if value is None: 

130 filtered_data = filtered_data.where(F.col(dimension).isNull()) 

131 else: 

132 filtered_data = filtered_data.where(F.col(dimension) == F.lit(value)) 

133 return filtered_data

136def _get_dimension_value_combinations(cube_dimensions: Union[Text, Sequence[Text]], 

137 eventlog: PysparkLog) -> List[Row]: 

138 dimension_value_combinations = ( 

139 eventlog 

140 .data_frame 

141 .groupBy(cube_dimensions) 

142 .agg(F.count(F.lit(1)).alias('count')) 

143 .where(F.col('count') > 1) 

144 .drop('count') 

145 .collect() 

146 ) 

147 return dimension_value_combinations
