20def build(eventlog: PysparkLog, 

21 make_start_end: bool = False, 

22 pm4py_params: bool = None) -> DfgResult: 

23 """ 

24 Counts the number of directly follows occurrences, i.e. of the form <...a,b...>, in an event log. 

25 

26 Parameters 

27 ---------- 

28 eventlog 

29 Log in the form of a pm4pyspark.log.PysparkLog 

30 make_start_end 

31 A boolean parameter that indicates whether or not to additionally create and return: 

32 - a dict with counts of the start activities 

33 - a dict with counts of the end activities 

34 These respectively are equivalent to what can be obtained with 

35 log.get_start_activities and log.get_end_activities. 

36 Calling this method with `make_start_end=True` is computationally more efficient than 

37 calling it with `make_start_end=False` and running log.get_start_activities and 

38 log.get_end_activities separately. 

39 pm4py_params 

40 Possible parameters passed to the algorithms: 

41 window -> the window size for the DFG. Defaults to 1. 

42 

43 Returns 

44 ------- 

45 Returns 

46 ------- 

47 result 

48 A pm4pyspark.dfg.frequency.DfgResult object consisting of: 

49 - result.dfg 

50 The DFG graph 

51 - start_activities 

52 A dict with the counts of how frequently each activity occurs at start of a trace, 

53 or `None` when `make_start_end=False` 

54 - end_activities 

55 A dict with the counts of how frequently each activity occurs at end of a trace, or 

56 or `None` when `make_start_end=False` 

57 """ 

58 if not isinstance(eventlog, PysparkLog): 

59 raise TypeError('eventlog argument is of type {} but should be a PysparkLog'.format(type(eventlog))) 

60 dfg_df = _build_spark_dataframe(eventlog, make_start_end, pm4py_params) 

61 dfg, end_activities = _process_spark_output(dfg_df.collect(), eventlog, make_start_end) 

62 start_activities = _get_start_activities(eventlog, make_start_end) 

63 return DfgResult(TYPE, dfg, start_activities, end_activities)

74def _build_spark_dataframe(eventlog: PysparkLog, 

75 make_start_end: bool, 

76 pm4py_params: Optional[Mapping]) -> DataFrame: 

77 if pm4py_params is None: 

78 pm4py_params = {} 

79 window_size = pm4py_params[WINDOW] if WINDOW in pm4py_params else DEFAULT_WINDOW 

80 relevant_columns = [eventlog.case_glue, eventlog.activity_key] 

81 spark_df = eventlog.get_and_cache_data() 

82 if 'timestamp_key' in dir(eventlog): 

83 sorting_key = eventlog.timestamp_key 

84 else: 

85 sorting_key = 'monotonic_row_id' 

86 spark_df = ( 

87 spark_df 

88 .withColumn(sorting_key, F.monotonically_increasing_id()) 

89 ) 

90 relevant_columns += [sorting_key] 

91 window = Window().partitionBy(eventlog.case_glue).orderBy(sorting_key) 

92 dfg_df = ( 

93 spark_df 

94 .select(relevant_columns) 

95 .withColumn(NEXT_ACTIVITY_KEY, F.lead(eventlog.activity_key, window_size).over(window)) 

96 ) 

97 if make_start_end: 

98 dfg_df = dfg_df.fillna(ARTIFICIAL_END, subset=[NEXT_ACTIVITY_KEY]) 

99 else: 

100 dfg_df = dfg_df.where(F.col(NEXT_ACTIVITY_KEY).isNotNull()) 

101 dfg_df = ( 

102 dfg_df 

103 .groupBy([eventlog.activity_key, NEXT_ACTIVITY_KEY]) 

104 .agg(F.count(F.lit(1)).alias('count')) 

105 ) 

106 return dfg_df

109def _process_spark_output(df_collection_result: List[Row], 

110 eventlog: PysparkLog, 

111 make_start_end: bool) -> Tuple[Dict[Text, int], Dict[Text, int]]: 

112 dfg = {(row[eventlog.activity_key], row[NEXT_ACTIVITY_KEY]): row['count'] 

113 for row in df_collection_result if row[NEXT_ACTIVITY_KEY] != ARTIFICIAL_END} 

114 end_activities = None 

115 if make_start_end: 

116 end_activities = {row[eventlog.activity_key]: row['count'] 

117 for row in df_collection_result if row[NEXT_ACTIVITY_KEY] == ARTIFICIAL_END} 

118 return dfg, end_activities
